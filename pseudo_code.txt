void execute_pipeline(char **cmd_paths, char ***args, t_pip *struc, int num_cmds, char *input_file, char *output_file)
{
	int		fd_in, fd_out;
	int		pipe_fds[2];
	int		i;
	pid_t	pid;

	// Ouvrir le fichier d'entrée
	fd_in = open(input_file, O_RDONLY);
	if (fd_in < 0)
		handle_error("open input_file");

	// Boucle pour toutes les commandes
	for (i = 0; i < num_cmds; i++)
	{
		// Créer un pipe, sauf pour la dernière commande
		if (i != num_cmds - 1)
			pipe(pipe_fds);

		pid = fork();
		if (pid == 0) // Processus enfant
		{
			if (i == 0)  // Première commande
			{
				dup2(fd_in, STDIN_FILENO); // Redirige l'entrée standard depuis file1
				close(fd_in);
				dup2(pipe_fds[1], STDOUT_FILENO); // Redirige la sortie vers le pipe
				close(pipe_fds[0]); // Fermer l'extrémité de lecture du pipe
				close(pipe_fds[1]);
			}
			else if (i == num_cmds - 1)  // Dernière commande
			{
				fd_out = open(output_file, O_WRONLY | O_CREAT | O_TRUNC, 0644);
				if (fd_out < 0)
					handle_error("open output_file");
				dup2(struc->pipe_tab[0], STDIN_FILENO); // Redirige l'entrée depuis le dernier pipe
				close(struc->pipe_tab[0]);
				dup2(fd_out, STDOUT_FILENO); // Redirige la sortie vers le fichier
				close(fd_out);
			}
			else  // Commandes intermédiaires
			{
				dup2(struc->pipe_tab[0], STDIN_FILENO); // Entrée depuis le pipe précédent
				dup2(pipe_fds[1], STDOUT_FILENO); // Sortie vers le pipe suivant
				close(struc->pipe_tab[0]);
				close(pipe_fds[1]);
			}

			// Exécuter la commande actuelle
			execve(cmd_paths[i], args[i], struc->env);
			ft_putstr_fd(args[i][0], STDERR_FILENO);
			ft_putstr_fd(": command not found\n", STDERR_FILENO);
			exit(EXIT_FAILURE);
		}
		else  // Processus parent
		{
			// Fermer les pipes inutilisés dans le parent
			if (i == 0)
				close(fd_in);
			else
				close(struc->pipe_tab[0]);

			if (i != num_cmds - 1)
				struc->pipe_tab[0] = pipe_fds[0]; // Enregistrer l'extrémité de lecture du pipe pour la prochaine commande
		}
	}
	// Attendre la fin de tous les processus enfants
	for (i = 0; i < num_cmds; i++)
		wait(NULL);
}
